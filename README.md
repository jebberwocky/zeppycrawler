# zeppycrawler
scrap https://www.zepp.co.jp/ and organize schedule data for email and csv

# default.ini
put a config deafult.ini file under the directory as same as main.py
format
```
[env]
stage = dev
debug = true
[email]
email =true
sender =mail@mail.com
to =mail@mail.com
apppwd =password
[csv]
csv =true
directory =./ 
```
